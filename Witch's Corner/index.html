                void Sort(int arr[], int size) {
                    for (int i = 1; i < size; i++) {
                        int key = arr[i];
                        int f = i;
                        while (f > 0 and arr[f - 1] > key) {
                            arr[f] = arr[f - 1];
                            f -= 1;
                        }
                        arr[f] = key;
                    }
                }
